#DecisionTree
library(caret)
library(mlbench)
library(rpart)      
library(rpart.plot)
data(Soybean)
df <- na.omit(Soybean)
set.seed(123)
idx <- createDataPartition(df$Class, p = 0.7, list = FALSE)
train <- df[idx, ]
test  <- df[-idx, ]
train$Class <- droplevels(train$Class)
test$Class  <- droplevels(test$Class)
model <- rpart(Class ~ ., data = train) 
rpart.plot(model)                       
preds <- predict(model, test, type = "class") 
confusionMatrix(preds, test$Class)

#Preprocessing
library(mlbench)
library(caret)
data(Soybean)
na_percent <- colMeans(is.na(Soybean))
Soybean <- Soybean[, na_percent < 0.2]
Soybean <- na.omit(Soybean)
dummies <- dummyVars(Class ~ ., data = Soybean)
Soybean <- data.frame(
  predict(dummies, Soybean),
  Class = Soybean$Class
)
features <- names(Soybean) != "Class"
Soybean[, features] <- scale(Soybean[, features])
set.seed(123)
train_index <- createDataPartition(Soybean$Class, p = 0.7, list = FALSE)
train_data <- Soybean[train_index, ]
test_data  <- Soybean[-train_index, ]
